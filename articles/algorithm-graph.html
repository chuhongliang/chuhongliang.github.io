直角三角形:

* 
* * 
* * * 
* * * * 
* * * * * 

for (let x = 1; x <= 5; x++) {
    let s = '';
    for (let y = 1; y <= x; y++) {
        s += `* `
    }
    console.log(s);
}

倒立直角三角形:

* * * * * 
* * * * 
* * * 
* * 
*

for (let x = 5; x >= 1; x--) {
    let s = '';
    for (let y = 1; y <= x; y++) {
        s += `* `
    }
    console.log(s);
}

等腰三角形:

     * 
    * * 
   * * * 
  * * * * 
 * * * * * 

for (let x = 1; x <= 5; x++) {
    let s = '';
    for (let z = 6 - x; z >= 1; z--) {
        s += ' '
    }
    for (let y = 1; y <= x; y++) {
        s += `* `
    }
    console.log(s);
}
